  }
   stage('terraform init') {
       steps {
           sh "terraform init"

    stage('terraform fmt'){

            sh "terraform fmt"
        }
    }
     stage('terraform plan') {
         steps{
             sh "terraform plan"
             
         }
     }
    
    
    stage ('terraform apply'){
        steps{
            sh "terraform apply"
        }
    }
  }
  
}
